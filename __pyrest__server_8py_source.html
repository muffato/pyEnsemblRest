66  def __get_json_answer(self, url, content_type=None):
67 
68  # Rate limiter
69  # FIXME: check whether it is ever used and delete otherwise
70  if self.last_headers is not None:
71  time_remaining = int(self.last_headers['x-ratelimit-reset'])
72  requests_remaining = int(self.last_headers['x-ratelimit-remaining'])
73  t = time_remaining * math.exp( -requests_remaining / time_remaining )
74  if t > .001:
75  print("sleeping", t, "seconds before calling", self.server_url)
76  time.sleep(t)
77 
78  #print("getting "+self.server_url+"/"+url+" with the content_type:"+content_type)
79  while True:
80  resp, content = self.http.request(self.server_url+"/"+url, method="GET", headers={"Content-Type":content_type})
81  if resp.status == 429:
82  print(self.server_url, "asked to wait", resp['retry-after'], "seconds", file=sys.stderr)
83  time.sleep(float(resp['retry-after']))
84  else:
85  break
86 
87  if resp.status not in self.__return_codes:
88  raise UnknownResponseCodeException(resp.status, resp, content)
89  if self.__return_codes[resp.status][0] != "OK":

96  def __build_rest_answer(self, new_object, allowed_formats, optional_params, accessor, url, kwargs={}):
97 
98  format = kwargs.pop('output_format', None)
99  if format is not None:
100  format = format.lower()
101  if format not in allowed_formats:
102  raise UnavailableFormatException(format)
103  format = None
104 
105  if len(kwargs):
106  ps = set(kwargs).intersection(optional_params)
107  pairs = []
108  for p in ps:
109  vv = kwargs[p]
110  if isinstance(vv, list):
111  pairs.extend( (p,_) for _ in vv)
112  else:
113  pairs.append( (p,vv) )
114  url = url + "?" + "&".join("%s=%s" % (p,urllib.parse.quote(str(v))) for (p,v) in pairs)
115 
116  content = self.__get_json_answer(url, self.__content_types.get(format, self.__content_types['json']))

120  return content
